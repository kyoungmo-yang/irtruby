estimates_examinee_score(score_per_items = false, item_nos = nil, examinees = nil)
  • Functionailities 
 * Estimates examinees' score
  • Parameters 
 * [score_per_items] : if this paramter is true, calculates score of each items for examinees. Otherwise, calculates total score for each examinees.
 * [item_nos] : if this parameter is specified, calculates score for the items. Otherwise, calculates score for all items
 * [examinees] : if this parameter is specified, calculates socre for the examinees. Otherwise, calculates score for all examinees.
  • Return

     # File irt.rb, line 112
112:   def estimates_examinee_score(score_per_items = false, item_nos = nil, examinees = nil)
113:     if  examinees
114:       s_examinees = examinees
115:     else
116:       s_examinees = @test.examinees
117:     end
118:     
119:     s_examinees.each do |examinee|
120:       theta = examinee.theta
121:       if(theta == nil)
122:         raise("Should be estimate score afeter estimates examinees' thetas")
123:       end
124:       
125:       if item_nos
126:         score = Irt_ruby.test_characteristic_curve([theta], item_nos)
127:         responses = examinee.responses[(item_nos[0]-1)..(item_nos[item_nos.size-1]-1)]
128:       else
129:         score = Irt_ruby.test_characteristic_curve([theta])
130:       end
131:       examinee.score = score[0]
132:       
133:       if(score_per_items)
134:         #scores of each items for an examinee
135:         unless item_nos
136:           item_nos = (1..(@test.number_of_items)).to_a
137:         end
138:         item_nos.each do |no|
139:           examinee.scores[@test.get_item_no(no-1)] = Irt_ruby.test_characteristic_curve([theta], [no])[0]
140:         end
141:       end
142:     end
143:   end
